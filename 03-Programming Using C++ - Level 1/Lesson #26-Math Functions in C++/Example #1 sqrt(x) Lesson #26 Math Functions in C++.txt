#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	double x = 64;
	
	cout << "Square root value of 64 : "<< sqrt(x) << endl; // gives 8
	cout << "Square root value of 50 : "<< sqrt(50) << endl; // gives 7.07107

	return 0;

}
