/*
Write a program to calculate rectangle area and print it on the screen.
Example Inputs:
10
20
Outputs:
200
*/

#include <iostream>

using namespace std;

int main()
{
	float area, a, b;

	cin >> a;
	cin >> b;

	area = a * b;

	cout << "\n" << area;

	return 0;

}
