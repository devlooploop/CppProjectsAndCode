#include <iostream>

using namespace std;

int ReadIntN(int &N)
{
	
	cout << "Enter the N Value \n";
	cin >> N;
	cout << "******************************\n";

	return N;
}

void PrintFromNto1(int N)
{
	int i = 1;
	while (i <= N)
	{
		cout << i << endl;
		
		i++;
	}
	
}

int main()
{
	int number;

	PrintFromNto1(ReadIntN(number));
	

	return 0;
}
