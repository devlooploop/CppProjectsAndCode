#include <iostream>

using namespace std;

int main()
{
    int Num1, Num2, Num3;

    cout << "Please Enter the Numbers\n";
    cin >> Num1;
    cin >> Num2;
    cin >> Num3;

    cout <<endl<< Num1 << " +" << endl;
    cout << Num2 << " +" << endl;
    cout << Num3 << endl;
    cout << "------------------" << endl;
    cout << "Total = " << Num1 + Num2 + Num3 << endl;

    return 0;
}
