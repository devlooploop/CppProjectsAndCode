#include <iostream>
#include <cmath>

using namespace std;

int main()
{

	cout << "Absolute value of -10 : " << abs(-10) << endl;   // gives 10
	cout << "Absolute value of  10 : " << abs(10) << endl;    // gives 10 

	return 0;
}
