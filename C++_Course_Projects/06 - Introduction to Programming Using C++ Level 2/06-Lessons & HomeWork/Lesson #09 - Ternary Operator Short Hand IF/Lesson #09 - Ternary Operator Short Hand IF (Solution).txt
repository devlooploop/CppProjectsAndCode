// Write a program to do the following:
// - Using Ternary Operator:
//	Program to check if the number is Positive or Negative.
// 	Program to check if the number is Zero, Positive or 
//   Negative. (Use nested ternary operators).

#include<iostream>
using namespace std;


int main() 
{ 	
	int Number = 0;
	string Result;

	Result = (Number == 0) ? "Zero": ((Number > 0) ? "Positive" : 
		"Negative");
	 
	cout << "Number is " << Result;

	return 0; 
}
