// Lesson #24 - Float Format (printf)

#include<iostream>
#include <cstdio>

int main() {
    float PI = 3.14159265;

    printf("Precision Spesification of %.*f\n", 1, PI); // Precision Spesification of 3.1
    printf("Precision Spesification of %.*f\n", 2, PI); //  Precision Spesification of 3.14
    printf("Precision Spesification of %.*f\n", 3, PI); //  Precision Spesification of 3.142
    printf("Precision Spesification of %.*f\n", 4, PI); //  Precision Spesification of 3.1416

    // Another way to do it  
    printf("Precision Specification of %.1f\n", PI); // Precision Spesification of 3.1
    

    float x = 7.0, y = 9.0;
    printf("\nThe float division is: %.3f / %.3f = %.3f \n\n", x, y, x / y); // The float division is: 7.000 / 9.000 = 0.778

    double d = 12.45;
    printf("The double value is: %.3f \n", d); // The double value is: 12.450
    printf("The double value is: %.4f \n", d); // The double value is: 12.4500

    return 0;
}
