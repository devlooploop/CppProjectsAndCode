/*
 - Write a program to store multiplication table 10x10 using 
	" 2 dimensional array size 10 x 10 " and print it's result. 
	using nested loops i & j,use printf() for formatting. 
	
	//int x[Rows][Cols]; ( 2 Dimensional array)
*/


#include <cstdio>
#include<iostream>

using namespace std;

int main() 
{ 
	//int x[Rows][Cols];
	int x[10][10]; 
	
	for (int i = 0; i < 10; i++)     
	{ 
		for (int j = 0; j < 10; j++)         
		{ 
			x[i][j] = (i + 1) * (j + 1);    //array is based 0 so we give i&j +1 as (i + 1) 
		}     
	} 
	
	for (int i = 0; i < 10; i++)     
	{ 
		for (int j = 0; j < 10; j++)         
		{             
			printf("%0*d ", 2, x[i][j]);         
		}         
		cout << endl;     
	} 
	
	return 0; 
}
