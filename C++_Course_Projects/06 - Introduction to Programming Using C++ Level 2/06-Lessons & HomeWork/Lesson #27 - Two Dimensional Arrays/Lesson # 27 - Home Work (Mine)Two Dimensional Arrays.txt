/*
 - Write a program to store multiplication table 10x10 using 
	" 2 dimensional array size 10 x 10 " and print it's result. 
	using nested loops i & j,use printf() for formatting. 
	
	//int x[Rows][Cols]; ( 2 Dimensional array)
*/


#include<iostream>

using namespace std;

void MultiplyTableFrom1To10()
{
	int Multiply[11][11] = {};

	for (int i = 1; i < 11; i++)
	{
 		for (int j = 1; j < 11; j++)
		{
			Multiply[i][j] = i * j;

			if ( ( Multiply[i][j] < 10))
			{
				printf("0%d", Multiply[i][j]);
				printf(" ");
			}
			else
			{
				printf("%d", Multiply[i][j]);
				printf(" ");
			}
			
		}
		printf("\n");
	}

}


int main() {

	//int x[Rows][Cols];
	
	MultiplyTableFrom1To10();


	return 0;
}
