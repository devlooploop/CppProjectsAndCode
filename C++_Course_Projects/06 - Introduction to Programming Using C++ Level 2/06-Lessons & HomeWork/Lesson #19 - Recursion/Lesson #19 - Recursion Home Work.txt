// -Using Recursion:
// . Program to print numbers from M down to N.
// . Program to calculate power N^M.

#include<iostream>
using namespace std;

void PrintNumbersFromNtoM(int N, int M)
{
	if (N <= M)
	{
		cout << N << endl;
		PrintNumbersFromNtoM(N + 1, M);
	}

}

int MyPowr(int Base, int Power)
{
	if (Power == 0)
	{
		return 1;
	}
	else
	{
		return	(Base * MyPowr(Base, Power - 1));
	}
	
}

int main()
{
	PrintNumbersFromNtoM(1,10);
	cout << endl;

	cout << MyPowr(2,4);

	return 0;
}
