// Problem #46 Print Letters from A to Z

/* Write a program to print all letters from A to Z
Outputs:
A ... Z
    */

#include <iostream>

using namespace std;

void PrintLettersAtoZ()
{
    for (int i = 65; i <= 90; i++)
    {
        cout << char(i) << endl;
    }
    
}

int main()
{
    PrintLettersAtoZ();

    return 0;
}
