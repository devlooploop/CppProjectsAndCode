/*
Write a program to read a number and print the sum of its digits.

Input:
1234

Output:
Sum of digits = 10

*/

#include<iostream>
#include<string>

using namespace std;

int ReadPositiveNumber(string Message) 
{ 
	int Number = 0 ;
	
	do 
	{ 
		cout << Message << endl;         
		cin >> Number; 
	} while (Number <= 0); 
	
	return Number;
} 

void PrintSumDigits(int Number) 
{ 
	int  Remainder = 0, Sum=0; 

	while (Number > 0) 
	{ 
		Remainder = Number % 10; 
		Number = Number / 10; 
		
		Sum += Remainder;
	} 
	cout << "Sum of digits = " << Sum << endl;
}


int main() 
{ 
  
	PrintSumDigits(ReadPositiveNumber("Please enter a positive number?"));
	
	return 0; 
}
