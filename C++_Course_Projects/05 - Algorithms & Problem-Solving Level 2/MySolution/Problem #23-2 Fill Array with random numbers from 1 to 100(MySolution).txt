// Problem #23/2 Fill Array with random numbers from 1 to 100

/*
Write a program to fill array with max size 100 
	with random numbers from 1 to 100.
Input:
10
Output:
Array Elements: 56 55 83 71 32 52 17 28 71 52

*/

#include <iostream>
#include <cstdlib>

using namespace std;

void ReadAndPrintRandomArrayNumbers(int arr[100], int& arrLength)
{
	cout << "Enter number of elements:\n";
	cin >> arrLength;

	cout << "Array Element:";

	for (int i = 0; i < arrLength; i++)
	{	
		arr[i] = rand() % 100;
		cout << arr[i] << " ";
	}
	cout << endl;
}

int main()
{
	// Function to generate a random number
	//Seeds the random number generator in C++, called only once    
	srand((unsigned)time(NULL));   
	
	int arr[100]{}, arrLength = 0;

	ReadAndPrintRandomArrayNumbers( arr, arrLength);

	return 0;
}
