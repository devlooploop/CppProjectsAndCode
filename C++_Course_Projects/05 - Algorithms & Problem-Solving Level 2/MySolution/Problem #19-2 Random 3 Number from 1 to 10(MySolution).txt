 /*
Write a program to print 3 Random Number from 1 to 10.

Output:
 9
 5
 3

*/

#include <iostream>
#include <cstdlib>

using namespace std; 


int PrintRandomNumber(int From, int To)
{
	 int Rand = ( rand() % (To - From + 1 ) ) + From;
	
	return Rand;
}


int main()
{
	srand(time(0));

	cout << PrintRandomNumber(1,10) << endl;
	cout << PrintRandomNumber(4,8)  << endl;
	cout << PrintRandomNumber(5,6)  << endl;

    return 0;
}
