// Problem #20-2 Random Small Letter, Capital Letter, Special Character, and Digit in order

/*

Write a program to print Random Small Letter, 
 Capital Letter, Special Character, 
   and Digit in order
 Output:
   i
   G
   $
   7
*/

#include <iostream>
#include <cstdlib>

using namespace std; 

enum enSmallLetter { From = 97, To = 122 };
enum enCapitalLetter { A = 65, B = 90 };
enum enSpecialCharacter { C = 32, D = 47 };
enum enDigit { E = 48, F = 57 };

char RandomNumber(int From , int To )
{
	
	// Function to generate a random number
	int randNum = rand() % (To - From + 1) + From;
	
	return (char)randNum;
}


int main()
{
	//Seeds the random number generator in C++, called only once    
	srand((unsigned)time(NULL));     

	cout << RandomNumber(enSmallLetter::From, enSmallLetter::To) << endl;
	cout << RandomNumber(enCapitalLetter::A , enCapitalLetter::B) << endl;
	cout << RandomNumber(enSpecialCharacter::C, enSpecialCharacter::D) << endl;
	cout << RandomNumber(enDigit::E, enDigit::F) << endl;


    return 0;
}
